rss_rating:
    id
    tag 所属新闻标签
    country 所属国家
    currency 所属国家主流货币
    rating 对所属国家货币评分

SQL: SQL: create table tag(id int(11) NOT NULL AUTO_INCREMENT, tag varchar(128) DEFAULT NULL, country varchar(128) DEFAULT NULL, currency varchar(128) DEFAULT NULL, rating varchar(128) DEFAULT NULL, PRIMARY KEY (id))DEFAULT CHARSET=utf8;
